CREATE DATABASE fusionpbx;

CREATE USER pbxuser WITH PASSWORD 'fusion';

ALTER ROLE pbxuser SET client_encoding TO 'utf8';

ALTER ROLE pbxuser SET default_transaction_isolation TO 'read committed';

ALTER ROLE pbxuser SET timezone TO 'CET';

GRANT ALL PRIVILEGES ON DATABASE fusionpbx TO pbxuser;

\q

sudo gpasswd -a www-data fusionpbx
